\begin{frame}
    \frametitle{What is Nix?}

    \begin{itemize}
        \item A Linux distro (NixOS). \faLinux\ {\tiny (also see NixBSD)}
        \noteitem{NixBSD is a fork to make a BSD equivalent of NixOS. Nix the package manager isn't available, though (somewhat surprising given that Darwin is mostly BSD-ish).}
        \item A pure functional package manager (Nix)\ldots\ \faLinux~\faApple\ {\tiny (technically Darwin)}
        \item \ldots based on a pure functional programming language \newline
        (also named Nix \emoji{face-with-rolling-eyes}). \faLinux~\faApple
        \noteitem{No Windows, but works under WSL. There's a Nix community module to install NixOS.}
        \item Reproducibility is baked into all three.
        \item Created by Eelco Dolstra in 2003.
        \begin{tcolorbox}{}
            \small
            ``The name \emph{Nix} is derived from the Dutch word \emph{niks}, meaning \emph{nothing}; build actions do not see anything that has not been explicitly declared as an input.'' \hfill{\tiny (\url{https://edolstra.github.io/pubs/nspfssd-lisa2004-final.pdf}, p.\ 81)}
        \end{tcolorbox}
        \noteitem{Fun fact: The logo was originally created for the Haskell logo competition, hence the lambdas. ``The snowflake motif is even more appropriate for Nix, because \emph{nix} is Latin for \emph{snow}.''}
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Some basic terminology}
    \framesubtitle{(\url{https://nixos.org/guides/how-nix-works/})}

    \begin{itemize}
        \item Packages are specified using \textbf{derivations}.
        \begin{itemize}
            \item pure functional (Nix the language)
            \item variants are easy
            \noteitem{Define the derivation as a function and pass arguments per variant.}
        \end{itemize}
        \item Packages are distributed via \textbf{channels}.
        \begin{itemize}
            \item e.g., \texttt{nixos-25.11}, \texttt{nixpkgs-25.11-darwin}, \texttt{nixpkgs-unstable}
            \item ``more than \num{120000} packages'' in \texttt{nixpkgs}
            \noteitem{This is the largest collection of any package manager I know of.}
        \end{itemize}
        \noteitem{cf.\ Debian stable, unstable.}
        \item Packages are installed locally in the \textbf{Nix store}.
        \begin{itemize}
            \item binary or source installs
            \item paths are hash based {\tiny (e.g., \texttt{/nix/store/b6gvzjyb2pg0kjfwrjmg1vfhh54ad73z-firefox-33.1/})}
            \noteitem{The hash is of all inputs used to build the package.}
            \item all dependencies are included under the same path
            \noteitem{Versioned accordingly. The obvious downside is storage usage, but eh\ldots}
            \item trivial to install multiple different versions
            \noteitem{Each version goes into a unique path and is isolated from all others. You can switch versions on the fly and specify the preferred version.}
            \item package upgrades and rollbacks are atomic
            \noteitem{Different inputs \(\Rightarrow\) different path.}
        \end{itemize}
    \end{itemize}
\end{frame}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%


\begin{frame}
    \frametitle{Deploying configurations and packages}

    \begin{itemize}
        \item NixOS:
        \begin{itemize}
            \item \texttt{/etc/nixos/configuration.nix}
            \item \texttt{nixos-rebuild switch}
            \item entire OS---including packages---is (re)built by the package manager!
            \noteitem{Your entire system specification becomes a single Nix derivation.}
        \end{itemize}
        \item Temporarily install/activate packages:
        \begin{itemize}
            \item \texttt{nix-shell -p firefox} {\tiny (old style)}
            \item \texttt{nix shell nixpkgs\#firefox} {\tiny (new style, using ``flakes'')}
            \item automatically using tools like \texttt{direnv}
        \end{itemize}
        \item Many users now use Nix \textbf{flakes}.
        \begin{itemize}
            \item effectively a collection of derivations wrapped in a Git repo
            \noteitem{I'm over-simplifying.}
            \item you can import flakes at any time
            \item building \texttt{flake.nix} \(\Rightarrow\) \texttt{flake.lock} ensures reproducibility
            \noteitem{cf.\ similar concept in other environments. It's a bit like running on stable Debian, but you can more easily update to newer packages.}
        \end{itemize}
        \item Many extension modules, e.g., \texttt{home-manager}, \texttt{nix-darwin}, \ldots.
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Nix + \texttt{direnv} = reproducibility superpowers}

    \begin{itemize}
        \item \texttt{direnv} lets you create per-directory reproducible environments. \tinyskip
        {\tiny (cf.\ rbenv, pyenv, phpenv)}
        \begin{itemize}
            \item \texttt{.envrc} contains shell code, run when you enter the directory
            \item dynamically load and unload environment variables {\tiny (subshell)}
        \end{itemize}
        \item To integrate with Nix:
        \begin{enumerate}
            \item add \texttt{use flake .} to your \texttt{.envrc}
            \item specify requirements in \texttt{flake.nix}
            \item \texttt{direnv allow .}
            \item \ldots profit!
        \end{enumerate}
        \noteitem{This a much more general solution than the language-specific tools. You can install \emph{any} dependencies, not just language modules. Everything resets when you leave (or deny) the directory.}
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{The installation landscape is slightly messy}
    \framesubtitle{(outside NixOS)}

    \begin{itemize}
        \item Official Nix installer is a bit bare bones.
        \item Determinate Systems created a ``batteries included'' installer:
        \begin{itemize}
            \item installed either ``vanilla'' or ``Determinate'' Nix {\tiny (= ``enterprise'' Nix)}
            \noteitem{This is somewhat analogous to Red Hat vs.\ Fedora, but different organisations.}
            \noteitem{The default switched from vanilla to Determinate in late 2025.}
            \item Determinate Nix clashes with \texttt{nix-darwin}
            \noteitem{As I discovered once when I inadvertently installed it.}
            \item recently (2026) dropped ``vanilla'' Nix support
        \end{itemize}
        \item Forks of the Determinate installer:
        \begin{itemize}
            \item ``official'' but experimental installer
            \item Lix installer {\tiny (\url{https://lix.systems})}
            \noteitem{I used to use Determinate, but have switched to Lix.}
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{The experience on macOS}

    \begin{itemize}
        \item Darwin requires the \texttt{nix-darwin} module.
        \begin{itemize}
            \item support for Homebrew, which supports the Mac App Store \tinyskip
            {\tiny (mainly useful for GUI apps and some fonts)}
            \item exposes a bunch of system-level settings
        \end{itemize}
        \item Settings configuration is somewhat patchy:
        \noteitem{But this is still a massive improvement over the previous state of affairs.}
        \begin{itemize}
            \item CLI configs are mostly straightforward {\tiny (via \texttt{home-manager})}
            \noteitem{Creating your own derivations for edge cases is relatively simple.}
            \item \texttt{nix-darwin} exposes some things but not all
            \item writing to the \texttt{defaults} database can be limited
            \noteitem{Probably not too dissimilar to the Windows Registry. Some items are opaque blobs, some are read only, some you write to and nothing happens, \ldots.}
            \item many apps still do their own thing
        \end{itemize}
        \item Package reproducibility is great! \tinyskip
        {\tiny(but Homebrew package versions can get ``stuck in time'')}
        \item External interference:
        \begin{itemize}
            \item system updates occasionally stomp on things {\tiny (mainly \texttt{/etc})}
            \noteitem{e.g., automount configuration.}
            \item OUDW regularly gets in the way \emoji{face-with-rolling-eyes}
        \end{itemize}
    \end{itemize}
\end{frame}
